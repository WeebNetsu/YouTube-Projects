	WSGIProcessGroup FlaskApp
	WSGIScriptAlias / /var/www/FlaskApp/flaskapp.wsgi

		Allow from all
	</Directory>
	Alias /static /var/www/FlaskApp/FlaskApp/static

		Order allow,deny
		Allow from all
	</Directory>
